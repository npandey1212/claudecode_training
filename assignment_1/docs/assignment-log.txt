  ---
  How Pydantic's Type System Works

  Every custom type hooks in via __get_pydantic_core_schema__. The dispatch chain is:

  1. _generate_schema_inner (_generate_schema.py:1007) calls match_type

  ---

  The closest real-world analogue is IPvAnyAddress (pydantic/networks.py:1174) — a string-like type that parses/validates and has
  custom serialization. Here's the template:

  # pydantic/xml_types.py  (or add to pydantic/types.py)

  from __future__ import annotations
  from typing import Any
  from xml.etree import ElementTree as ET
  from pydantic import GetCoreSchemaHandler
  from pydantic_core import core_schema


  class XMLDatatype:
      """A Pydantic type that accepts and validates XML strings."""

      def __init__(self, xml_string: str) -> None:
          self._xml = xml_string
          self._element = ET.fromstring(xml_string)  # raises ParseError if invalid

      @property
      def root_tag(self) -> str:
          return self._element.tag

      def __str__(self) -> str:
          return self._xml

      def __repr__(self) -> str:
          return f"XMLDatatype(root_tag={self.root_tag!r})"

      @classmethod
      def _validate(cls, value: Any) -> XMLDatatype:
          if isinstance(value, cls):
              return value
          if isinstance(value, str):
              try:
                  return cls(value)
              except ET.ParseError as e:
                  raise ValueError(f"Invalid XML: {e}") from e
          raise ValueError(f"Expected str or XMLDatatype, got {type(value)}")

      @classmethod
      def __get_pydantic_core_schema__(
          cls,
          source: type[Any],
          handler: GetCoreSchemaHandler,
      ) -> core_schema.CoreSchema:
          # Accept str in lax mode, only XMLDatatype in strict mode
          return core_schema.lax_or_strict_schema(
              lax_schema=core_schema.no_info_plain_validator_function(
                  cls._validate,
                  serialization=core_schema.to_string_ser_schema(),
              ),
              strict_schema=core_schema.is_instance_schema(
                  cls,
                  serialization=core_schema.to_string_ser_schema(),
              ),
          )
